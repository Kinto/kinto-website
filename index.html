Deploy once, reuse

Deploy once, reuse it for many apps

var kinto = new Kinto({remote: "https://kinto.dev.mozaws.net/v1"});
var tasks = kinto.collection("tasks");

tasks.create({label: "Wake up early"});
tasks.sync()
.then(function(res) {
    console.log("Done!");
});
